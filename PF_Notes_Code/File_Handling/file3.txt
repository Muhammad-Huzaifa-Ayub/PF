#INCLUDE <IOSTREAM>
#INCLUDE <CSTDLIB>
#INCLUDE <FSTREAM>

USING NAMESPACE STD;

INT MAIN() 
{
	STRING NUMFILE = "VALUES.TXT";
	OFSTREAM OFILE;
	OFILE.OPEN(NUMFILE);
	//IF (!OFILE.IS_OPEN())
	INT J=0;	
	WHILE(J<10)
	{
		OFILE << J << '\T' << RAND()%850+150 << ENDL;
		J = J + 1;
	}
	
	OFILE.CLOSE();
	
   RETURN 0;
}
